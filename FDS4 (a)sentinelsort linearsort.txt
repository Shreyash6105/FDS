def linear_search(arr, key):
    return key in arr

def sentinel_search(arr, key):
    last = arr[-1]
    arr[-1] = key
    i = 0
    while arr[i] != key:
        i += 1
    arr[-1] = last
    return i < len(arr) - 1 or arr[-1] == key
